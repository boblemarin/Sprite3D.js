 31 /**
 32  * Creates an instance of Sprite3D
 33  *
 34  * @constructor
 35  * @author boblemarin
 36  * @this {Sprite3D}
 37  * @param {Object} element The DOM element to wrap the Sprite3D object around. When no element is provided, a empty div is created and added to the document.
 38  */
 39 function Sprite3D(element) {
 40     // private variables
 41     var p,s,
 42     rx,
 43     ry,
 44     rz,
 45     i,
 46     alpha = 1,
 47     listeners = {};
 48 
 49     // create an empty <div> if no element is provided
 50     if (element == null)
 51     {
 52         element = document.createElement("div");
 53     }
 54 
 55     // prepare for 3D positionning
 56     element.style.webkitTransformStyle = "preserve-3d";
 57     element.style.margin = "0px";
 58     element.style.padding = "0px";
 59     element.style.position = "absolute";
 60 
 61     // trigger hardware acceleration even if no property is set
 62     element.style.webkitTransform = "translateZ(0px)";
 63 
 64     // debug style
 65     //element.style.border = '1px solid red';
 66     this.domElement = element;
 67     this.style = element.style;
 68     this.children = [];
 69 }

556 /**
557  * Applies position and rotation values.
558  * @return {Sprite3D} The reference to this Sprite3D object
559  */
560 Sprite3D.prototype.update = function() {
561     p = "translate3d(" + (this.x - this.regX) + "px," + (this.y - this.regY) + "px," + (this.z - this.regZ) + "px) ";
562     rx = "rotateX(" + this.rotationX + "deg) ";
563     ry = "rotateY(" + this.rotationY + "deg) ";
564     rz = "rotateZ(" + this.rotationZ + "deg) ";
565 	s = "scale3d(" + this.scaleX + ", " + this.scaleY + ", " + this.scaleX + ") ";
566 
567     if (this.rotateFirst)
568     //this.style.webkitTransform = rx + ry + rz + p;
569     this.style.webkitTransform = rz + ry + rx + p + s;
570     else
571     this.style.webkitTransform = p + rx + ry + rz + s;
572 
573     return this;
574 
575     /*
576 	future version, more flexible :
577 	
578 	var transformString = "rx ry rz p";
579 	var tt = transformString;
580 	tt = tt.replace( "p", p );
581 	tt = tt.replace( "rx", rx );
582 	tt = tt.replace( "ry", ry );
583 	tt = tt.replace( "rz", rz );
584 	
585 	this.style.webkitTransform = tt;
586 	
587 	*/
588 };

590 /**
591  * Applies position and rotation values, as well as opacity and size.
592  * @return {Sprite3D} The reference to this Sprite3D object
593  */
594 Sprite3D.prototype.updateAll = function() {
595     this.update();
596     this.style.opacity = this.alpha;
597     this.style.width = this.width + "px";
598     this.style.height = this.height + "px";
599     return this.update();

603 /**
604  * Calls the update() method on every child of the Sprite3D object.
605  * @return {Sprite3D} The reference to this Sprite3D object

607 Sprite3D.prototype.updateChildren = function() {
608     for (i = 0; i < this.numChildren; i++) {

610     }
611     return this;

625 /**
626  * Updates itself, then calls the update() method on every child of the Sprite3D object.
627  * @param {boolean} recursive If set to true, make the update call recursive, update every child's children
628  * @return {Sprite3D} The reference to this Sprite3D object
629  */
630 Sprite3D.prototype.updateWithChildren = function(recursive) {
631     this.update();
632 
633     for (i = 0; i < this.numChildren; i++) {
634         if (recursive)
635         this.children[i].updateWithChildren(recurse);
636         else
637         this.children[i].update();
638     }
639 
640     return this;
641 };

668  * Removes the nth Sprite3D object from this Sprite3D children.
669  * @param {number} n The index of the Sprite3D object to remove
670  * @return {Sprite3D} The reference to the removed Sprite3D object.
671  */
672 Sprite3D.prototype.removeChildAt = function(n) {
673     --this.numChildren;
674     this.domElement.removeChild(this.children[n].domElement);
675     return this.children.splice(n, 1)[0];
676 };

678 // returns the child associated with the provided Dom element
679 // use this when listening to user input events (using addEventListener)
680 /**
681  * Finds and return the Sprite3D object associated with the provided DOM element
682  * @param {Object} element The DOM element
683  * @return {Sprite3D} The reference to the associated Sprite3D object. Returns null if no relevant Sprite3D object was found
684  */
685 Sprite3D.prototype.findFromDOMElement = function(element) {
686     for (i = 0; i < this.numChildren; i++) {
687         if (element == this.children[i].domElement) return this.children[i];
688     }
689     return null;
690 };

692 /**
693  * Adds an event listener to the DOM element for the provided event id.
694  * @param {String} event The name of the event to watch
695  * @param {Function} callback The callback function
696  * @return {Sprite3D} The reference to this Sprite3D object
697  */
698 Sprite3D.prototype.addEventListener = function(event, callback) {
699     // old way, not so satisfying (does not bring the sprite's reference to the callback funcion)
700     //this.domElement.addEventListener( event, callback );
701     // experimental version, allows to get the reference of the target Sprite3D,
702     // but requires a little bit more work on the removeEventListener method
703     var sprite = this;
704     this.domElement.addEventListener(event,
705     function(e) {
706         callback(e, sprite);
707     });
708     return this;
709 };

724  * Creates a centered empty HTML div element to be used as root container for the other Sprite3D objects.
725  * @return {Sprite3D} The created Sprite3D object
726  */
727 Sprite3D.createCenteredContainer = function() {
728     var c = document.createElement('div');
729     var s = c.style;
730     s.webkitPerspective = "800";
731     s.webkitPerspectiveOrigin = "0 0";
732     s.webkitTransformOrigin = "0 0";
733     s.webkitTransform = "translateZ(0px)";
734     s.position = "absolute";
735     s.top = "50%";
736     s.left = "50%";
737     s.margin = "0px";
738     s.padding = "0px";
739     //s.border = "1px solid red";
740     document.body.appendChild(c);
741 
742     return new Sprite3D(c);
743 };
